####Commands for generating PCA

#Read experimental design for Alta
RNASEQ_design_Alta <- read.csv("RNASEQ_design_Alta.csv",header = TRUE,row.names = 1)

#Read the count data for Alta
##using csv2 command
Count_table_Alta_gene_level <- read.csv2("Count table_Alta_gene_level.csv", header = TRUE, sep = ";", row.names = 1)
#set factor levels

RNASEQ_design_Alta$GenoTissue <- factor(RNASEQ_design_Alta$GenoTissue)
RNASEQ_design_Alta$Treatment_D_cold <- factor(RNASEQ_design_Alta$Treatment_D_cold)

#Creat a deseq object and import the count data and sample information
dds_Alta <- DESeqDataSetFromMatrix(countData = Count_table_Alta_gene_level,colData = RNASEQ_design_Alta, design =~Treatment_D_cold)

dds_Alta$Treatment_D_cold <- factor(dds_Alta$Treatment_D_cold, levels = c("0H","42D"))

#PCA plots
#variance stabilizing transformation
vsd_Alta <- vst(dds_Alta,blind = TRUE)

#Use transformed values to generate a PCA plot
##Generating PCA with ggplot2
pcaData_Alta <- plotPCA(vsd_Alta,intgroup=c("GenoTissue", "Treatment_D_cold"), returnData=TRUE)
percentVar <- round(100 * attr(pcaData_Alta, "percentVar"))

PCAplot_Alta <- ggplot(pcaData_Alta, aes(PC1, PC2, color=Treatment_D_cold, shape=GenoTissue)) +
  geom_point(size=2.5, alpha=1) +
  scale_shape_manual(values=c(6, 4, 2, 5)) +
  xlab(paste0("PC1: ",percentVar[1],"% variance")) +
  ylab(paste0("PC2: ",percentVar[2],"% variance")) + 
  coord_fixed(xlim = c(-80,80),ylim = c(-50, 50))+
  theme(panel.background = element_blank(), panel.border = element_rect(color = "grey", fill = NA, size = 0.05))+ 
theme_bw()+
theme(legend.position = c(0.5, 0.5)) + 
theme(legend.background = element_rect(size=0.5, linetype="solid", colour ="black")) + 
theme(legend.title = element_text(size=8))+ 
theme(legend.text= element_text(size=8))

PCAplot_Alta

###save file as pdf
ggsave('Alta PCA plot.pdf', width=7.78, height=2.67, plot=PCAplot_Alta, device = "pdf", path="./", dpi=600)

##Save file as svg
library(svglite)
ggsave('Alta PCA plot.svg', width=7.78, height=2.67, plot=PCAplot_Alta, device = "svg", path="./", dpi=600)
