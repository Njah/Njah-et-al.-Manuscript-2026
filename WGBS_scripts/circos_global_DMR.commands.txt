####The following command is found in circos.conf folder. Each circos.conf required for generating each plot is separated by **********

########For DMR scatter AL_FL_NL_CG

# circos.conf

<<include colors_fonts_patterns.conf>>

<<include ideogram.conf>>
<<include ticks.conf>>

<image>
file* = circos.ALFLNL_CG.scatter.8.svg
<<include etc/image.conf>>
</image>

karyotype   = data/karyotype/Karyotype.F.vesca.txt

chromosomes_units           = 1200000
chromosomes_display_default = yes

# Parameters used by dynamic r1/r0 expressions to calculate track position.
# This assumes that each track has the same width.
t0 = 0.98 # outer track radius
tw = 0.08 # track width
ts = 0.03 # track spacing

###plots


<plots>

# global parameters for each track
fill_color = white
show = yes
type = scatter
max  = 1
min  = -1
stroke_thickness = 1
stroke_color     = black
glyph            = circle
glyph_size       = 8


<plot>
file = data/DMRsBinsALCG.txt
r1   = 0.99r
r0   = 0.89r
color       = orange
fill_color  = orange
background_color = vvlgrey

<rules>
<rule>
condition    = var(value) > 0.1
color        = dred
fill_color   = dred
</rule>
<rule>
condition    = var(value) < -0.1
color        = dblue
fill_color   = dblue
</rule>
</rules>
</plot>

<plot>

color        = dred

</rule>
<rule>
condition    = var(value) < -0.1

</rules>
</plot>

<plot>
