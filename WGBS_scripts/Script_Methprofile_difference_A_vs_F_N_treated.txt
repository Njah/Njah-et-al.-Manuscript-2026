library("DMRcaller")

AK42D.methylationData <- readBismark("AK42D.CX_report.txt.gz")

FK42D.methylationData <- readBismark("FK42D.CX_report.txt.gz")

NK42D.methylationData <- readBismark("NK42D.CX_report.txt.gz")

AL42D.methylationData <- readBismark("AL42D.CX_report.txt.gz")

FL42D.methylationData <- readBismark("FL42D.CX_report.txt.gz")

NL42D.methylationData <- readBismark("NL42D.CX_report.txt.gz")

#Store data from all files in a GRanges object

methylationDataList_A_F_N <- GRangesList("AK42D" = AK42D.methylationData, "AL42D" = AL42D.methylationData,"FK42D" = FK42D.methylationData,"FL42D" = FL42D.methylationData,"NK42D" = NK42D.methylationData, "NL42D" = NL42D.methylationData)


#####  Chromosome 1

#Compute methylation profile for Fvb1

regions_A_F_N.Fvb1 <- GRanges(seqnames = Rle("Fvb1"), ranges = IRanges(1,24253023))

# compute low resolution in CG context using a window of 50k
# compute low resolution  for AK42D
profileAK42D.CG.Fvb1<- computeMethylationProfile(methylationDataList_A_F_N[["AK42D"]],regions_A_F_N.Fvb1,windowSize = 50000,context = "CG")

# compute low resolution  for FK42D
profileFK42D.CG.Fvb1<- computeMethylationProfile(methylationDataList_A_F_N[["FK42D"]],regions_A_F_N.Fvb1,windowSize = 50000,context = "CG")

# compute low resolution  for NK42D
profileNK42D.CG.Fvb1<- computeMethylationProfile(methylationDataList_A_F_N[["NK42D"]],regions_A_F_N.Fvb1,windowSize = 50000,context = "CG")

# compute low resolution  for AL42D
profileAL42D.CG.Fvb1<- computeMethylationProfile(methylationDataList_A_F_N[["AL42D"]],regions_A_F_N.Fvb1,windowSize = 50000,context = "CG")

# compute low resolution  for FL42D
profileFL42D.CG.Fvb1<- computeMethylationProfile(methylationDataList_A_F_N[["FL42D"]],regions_A_F_N.Fvb1,windowSize = 50000,context = "CG")

# compute low resolution  for NL42D
profileNL42D.CG.Fvb1<- computeMethylationProfile(methylationDataList_A_F_N[["NL42D"]],regions_A_F_N.Fvb1,windowSize = 50000,context = "CG")


#####   1. CG Fvb1   #############


#Proportion difference in meth profile between AK42D and FK42D in Fvb1 CG context
PropdiffAKFK42D.CG.Fvb1 <- profileAK42D.CG.Fvb1$Proportion - profileFK42D.CG.Fvb1$Proportion

#Create a new file for appending methprop difference
profileAKFK42D.CG.Fvb1 <- profileAK42D.CG.Fvb1

#Appending methylation proportion difference
profileAKFK42D.CG.Fvb1$PropdiffAKFK42D.CG.Fvb1 <- PropdiffAKFK42D.CG.Fvb1

write.csv(profileAKFK42D.CG.Fvb1, "profileAKFK42D.CG.Fvb1.csv")

#Proportion difference in meth profile between AK42D and NCGR1363 in Fvb1 CG context
PropdiffAKNK42D.CG.Fvb1 <- profileAK42D.CG.Fvb1$Proportion - profileNK42D.CG.Fvb1$Proportion

#Create a new file for appending methprop difference
profileAKNK42D.CG.Fvb1 <- profileAK42D.CG.Fvb1

#Appending methylation proportion difference
profileAKNK42D.CG.Fvb1$PropdiffAKNK42D.CG.Fvb1 <- PropdiffAKNK42D.CG.Fvb1


write.csv(profileAKNK42D.CG.Fvb1, "profileAKNK42D.CG.Fvb1.csv")


#Proportion difference in meth profile between AL42D and FL42D in Fvb1 CG context
PropdiffALFL42D.CG.Fvb1 <- profileAL42D.CG.Fvb1$Proportion - profileFL42D.CG.Fvb1$Proportion

#Create a new file for appending methprop difference
profileALFL42D.CG.Fvb1 <- profileAL42D.CG.Fvb1

#Appending methylation proportion difference
profileALFL42D.CG.Fvb1$PropdiffALFL42D.CG.Fvb1 <- PropdiffALFL42D.CG.Fvb1

write.csv(profileALFL42D.CG.Fvb1, "profileALFL42D.CG.Fvb1.csv")

#Proportion difference in meth profile between AL42D and NCGR1363 in Fvb1 CG context
PropdiffALNL42D.CG.Fvb1 <- profileAL42D.CG.Fvb1$Proportion - profileNL42D.CG.Fvb1$Proportion

#Create a new file for appending methprop difference
profileALNL42D.CG.Fvb1 <- profileAL42D.CG.Fvb1

#Appending methylation proportion difference
profileALNL42D.CG.Fvb1$PropdiffALNL42D.CG.Fvb1 <- PropdiffALNL42D.CG.Fvb1

###Export File
write.csv(profileALNL42D.CG.Fvb1, "profileALNL42D.CG.Fvb1.csv")

###Read edited file containing AK vs FK and NK proportion values

Final_profileAKFKNK42D.CG.Fvb1 <- read.csv("profileAKFKNK42D.CG.Fvb1.csv")

##Generate plots
library(ggplot2)

Plot_Final_profileAKFKNK42D.CG.Fvb1 <- ggplot(Final_profileAKFKNK42D.CG.Fvb1, aes(x=end, y=Propdiff.CG.Fvb1)) + geom_line(aes(color = group, alpha = 0.2))+ theme(panel.background = element_blank())+ scale_color_manual (values = c ("AK_FK" = "cyan", "AK_NK" = "red"))+ scale_y_continuous(limits = c(-0.5, 0.5), breaks = c(-0.5, -0.3, -0.1, 0, 0.1, 0.3, 0.5))+ scale_x_continuous(limits = c(0, 25000000), breaks = c(0, 5000000, 10000000, 15000000, 20000000, 25000000))+theme(panel.background = element_blank())+ geom_vline(xintercept = 0, color = "grey")+ geom_hline(yintercept=0, color = "grey")+xlab("chromosome coordinates") + ylab("Proportion difference")+ labs(title = "Methylation difference between AK and FK NK CG 42D  chr1") + theme(plot.title = element_text(hjust=0.5))
